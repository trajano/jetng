31  /**
32   * Processes the {@link Emit} and {@link Emits} annotations.
33   */
34  @SupportedAnnotationTypes({ "net.trajano.jetng.apt.Emits", "net.trajano.jetng.apt.Emit" })
35  @SupportedSourceVersion(SourceVersion.RELEASE_5)
36  public class EmitsProcessor extends AbstractProcessor {
37  
38      /**
39       * Known element types.
40       */
41      private static final Set<String> KNOWN_ELEMENT_TYPES;
42  
43      /**
44       * Logger.
45       */
46      private static final Logger LOG = Logger.getLogger("net.trajano.jetng.apt", "META-INF/Messages");
47  
48      static {
49          KNOWN_ELEMENT_TYPES = Collections.unmodifiableSet(new HashSet<String>(Arrays.asList(TypeElement.class.getName(), Element.class.getName(), VariableElement.class.getName(), ExecutableElement.class.getName(), PackageElement.class.getName())));
50      }
51  
52      /**
53       * Annotation processors.
54       */
55      private final Set<String> annotationProcessors = new TreeSet<String>();
56  
57      /**
58       * Elements processed.
59       */
60      private final Set<TypeElement> elementCollection = new HashSet<TypeElement>();
61  
62      /**
63       * {@inheritDoc}
64       */
65      @Override
66      public boolean process(final Set<? extends TypeElement> annotations,
67              final RoundEnvironment roundEnv) {
68  
69          for (final Element element : roundEnv.getElementsAnnotatedWith(Emits.class)) {
70              final TypeElement typeElement = (TypeElement) element;
71              for (final Emit emit : typeElement.getAnnotation(Emits.class)
72                      .value()) {
73                  processEmit(emit, typeElement);
74              }
75          }
76          for (final Element element : roundEnv.getElementsAnnotatedWith(Emit.class)) {
77              final TypeElement typeElement = (TypeElement) element;
78              processEmit(typeElement.getAnnotation(Emit.class), typeElement);
79          }
80          if (roundEnv.processingOver() && !roundEnv.errorRaised()) {
81              try {
82                  final PrintWriter metaWriter = new PrintWriter(processingEnv.getFiler()
83                          .createResource(StandardLocation.SOURCE_OUTPUT, "", "META-INF/services/javax.annotation.processing.Processor", elementCollection.toArray(new Element[0]))
84                          .openWriter());
85                  for (final String processor : annotationProcessors) {
86                      metaWriter.println(processor);
87                  }
88                  metaWriter.close();
89              } catch (final IOException e) {
90                  final String msg = MessageFormat.format(LOG.getResourceBundle()
91                          .getString("ioerror"), "META-INF/services/javax.annotation.processing.Processor", e.getMessage());
92                  processingEnv.getMessager()
93                  .printMessage(Kind.ERROR, msg);
94                  throw new IllegalStateException(msg, e);
95              }
96          }
97          return true;
98      }
99  
100     /**
101      * Handle the processing of an emit annotation.
102      *
103      * @param emit
104      *            emit annotation
105      * @param typeElement
106      *            type element
107      * @throws IOException
108      */
109     private void processEmit(final Emit emit,
110             final TypeElement typeElement) {
111 
112         elementCollection.add(typeElement);
113         String elementClassUsedInArgumentsClassConstructor = null;
114         for (final Element element : typeElement.getEnclosedElements()) {
115             if (ElementKind.CONSTRUCTOR == element.getKind()) {
116                 final ExecutableElement executableElement = (ExecutableElement) element;
117                 final List<? extends VariableElement> parameters = executableElement.getParameters();
118                 if (parameters.size() != 1 && KNOWN_ELEMENT_TYPES.contains(parameters.get(0)
119                         .asType()
120                         .toString())) {
121                     continue;
122                 }
123                 if (elementClassUsedInArgumentsClassConstructor != null) {
124                     final String msg = MessageFormat.format(LOG.getResourceBundle()
125                             .getString("manyconstructor"),
126                             typeElement.asType()
127                             .toString());
128                     processingEnv.getMessager()
129                     .printMessage(Kind.ERROR, msg);
130                     throw new IllegalStateException(msg);
131                 }
132                 elementClassUsedInArgumentsClassConstructor = parameters.get(0)
133                         .asType()
134                         .toString();
135             }
136         }
137         if (elementClassUsedInArgumentsClassConstructor == null) {
138             final String msg = MessageFormat.format(LOG.getResourceBundle()
139                     .getString("noconstructor"),
140                     typeElement.asType()
141                     .toString());
142             processingEnv.getMessager()
143             .printMessage(Kind.ERROR, msg);
144             throw new IllegalStateException(msg);
145         }
146         final EmitModel model = new EmitModel();
147         if ("".equals(emit.packageName())) {
148             model.setPackageName(((PackageElement) typeElement.getEnclosingElement()).getQualifiedName()
149                     .toString());
150         } else {
151             model.setPackageName(emit.packageName());
152         }
153         model.setGeneratorClassName(emit.template());
154         model.setClassName(emit.processorName());
155         model.setArgumentClassName(typeElement.asType()
156                 .toString());
157         model.setQualifiedNameMethod(emit.qualifiedNameMethod());
158         model.setArgumentConstructorParameterType(elementClassUsedInArgumentsClassConstructor);
159         model.setSupportedAnnotationTypes(emit.supportedAnnotationTypes());
160         model.setSupportedSourceVersion("SourceVersion." + emit.supportedSourceVersion());
161         try {
162 final EmitterGenerator generator = new EmitterGenerator();
163             annotationProcessors.add(model.getQualifiedName());
164             final PrintWriter writer = new PrintWriter(processingEnv.getFiler()
165                     .createSourceFile(model.getQualifiedName())
166                     .openWriter());
167             generator.generate(model, writer);
168             writer.close();
169         } catch (final IOException e) {
170             final String msg = MessageFormat.format(LOG.getResourceBundle()
171                     .getString("ioerror"),
172                     typeElement.asType()
173                     .toString(),
174                     e.getMessage());
175             processingEnv.getMessager()
176             .printMessage(Kind.ERROR, msg);
177             throw new IllegalStateException(msg, e);
178         }
179     }
